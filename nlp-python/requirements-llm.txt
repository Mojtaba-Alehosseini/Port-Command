# NLP tooling + LLM sidecar dependencies — Port Command Genova.

# PYTHON 3.10 or 3.11 ONLY (never 3.12+). Installed into the primary venv:
#
#   py -3.11 -m venv .venv
#   .venv/Scripts/pip install -r requirements-llm.txt

# This is the interpreter the pre-commit hook runs pytest under, and the one the
# ontology converter (task 02) and the Flask LLM sidecar (task 13) use. It is a
# SEPARATE process from the Rasa service, which runs on Python 3.10 out of
# .venv-rasa (see requirements-rasa.txt) because rasa 3.6.21 forbids 3.11.

# Force the CPU PyTorch wheel (the sidecar runs CPU-first; see planning/13). On
# Windows PyPI already serves a CPU-only torch, but this line makes the CPU build
# explicit and reproducible on Linux/CI too. `torch==2.5.1` matches `2.5.1+cpu`.
--extra-index-url https://download.pytorch.org/whl/cpu

# --- Ontology converter (task 02) ---
# owlready2 is used only as an OWL *parser* by ontology_to_assets.py (no reasoner).
owlready2==0.46

# --- Flask LLM sidecar (task 13) ---
# Phi-4-mini's model class landed in transformers 4.49; it also ships custom
# modeling code on the Hub (loaded with trust_remote_code=True). torch is pinned
# to the CPU build via the --extra-index-url above (~200 MB vs the multi-GB CUDA
# wheel). accelerate provides device_map. These are ~2.5 GB installed and are NOT
# needed to author/train/gate Rasa (task 12) — that is why task 12 deferred them.
Flask==2.3.3
# Flask 2.3.x predates Werkzeug 3.0 and uses APIs Werkzeug 3 removed; an unpinned
# install pulls Werkzeug 3.x (a latent runtime break). Pin the intended pairing.
Werkzeug==2.3.8
transformers==4.49.0
torch==2.5.1
accelerate==1.3.0

# --- Fast CPU backend: ONNX Runtime GenAI INT4 (task 13b, LLM_QUANT=onnx) ---
# THE fast CPU path. Phi-4-mini int4 via ONNX Runtime GenAI is dramatically faster
# than the fp16/bf16 default on this CPU (i7-8850H, AVX2, no AVX-512), where fp16 is
# ~9 min/explanation (task 13) — unusable in-game. Prebuilt CPU wheel: no CUDA, no
# compile (unlike the bnb/llama-cpp optionals below). model.py imports
# onnxruntime_genai LAZILY, so `import llm_sidecar.*` and the pytest suite stay
# hermetic. Weights are a one-time ~4.9 GB manual download OUTSIDE the repo; then
# set LLM_QUANT=onnx or let start_llm.{sh,bat} auto-select it. See README §Performance.
#   huggingface-cli download microsoft/Phi-4-mini-instruct-onnx --include "cpu_and_mobile/cpu-int4-rtn-block-32-acc-level-4/*" --local-dir "%USERPROFILE%\.cache\port-command-genova\Phi-4-mini-instruct-onnx"
onnxruntime-genai==0.14.1
onnxruntime==1.27.0
